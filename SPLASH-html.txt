<script>
(function(){
  var splash = document.getElementById('ledgerSplash');
  if(!splash) return;

  // SHOW ONCE: if they've seen it before, hide immediately and stop.
  // To make it show EVERY visit instead, delete these 4 lines.
  if(localStorage.getItem('ledgerSplashSeen')){
    splash.classList.add('is-done');
    return;
  }

  // Prevent background scroll while splash is up.
  document.body.style.overflow = 'hidden';

  function dismiss(){
    splash.classList.add('is-hiding');
    localStorage.setItem('ledgerSplashSeen', '1');   // remember for next time
    document.body.style.overflow = '';
    setTimeout(function(){ splash.classList.add('is-done'); }, 700);
  }

  // Auto-fade after 4 seconds.
  var timer = setTimeout(dismiss, 4000);

  // Tap anywhere (or the skip text) to dismiss early.
  splash.addEventListener('click', function(){ clearTimeout(timer); dismiss(); });
})();
</script>
